``` c
char * gcdOfStrings(char * str1, char * str2){
    int len1 = strlen(str1);
    int len2 = strlen(str2);
    
    if(len1 < len2)
    {
        return gcdOfStrings(str2, str1);
    }
    
    for(int i = 0; i < len2; i++)
    {
        if(str1[i] != str2[i])
        {
            return "";
        }
    }
    
    if(len1 == len2)
    {
        return str1;
    }
    
    return gcdOfStrings(str1 + len2, str2);
}
```
