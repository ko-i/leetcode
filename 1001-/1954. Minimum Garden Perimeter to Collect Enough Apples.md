``` rust
impl Solution {
    pub fn minimum_perimeter(mut needed_apples: i64) -> i64 {
        let mut n = 0;
        while needed_apples > 0 {
            n += 1;
            needed_apples -= n * n * 12;
        }
        
        n << 3
    }
}
```
