    pub fn kth_largest_number(mut nums: Vec<String>, k: i32) -> String {
        nums.select_nth_unstable_by(k as usize - 1, |a, b| {
            match a.len().cmp(&b.len()) {
                Less => Greater,
                Equal => b.cmp(a),
                Greater => Less,
            }
        });
        
        nums[k as usize - 1].to_string()
    }
}
