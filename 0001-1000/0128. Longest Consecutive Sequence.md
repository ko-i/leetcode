``` rust
use std::collections::HashSet;

impl Solution {
    pub fn longest_consecutive(nums: Vec<i32>) -> i32 {
        if nums.len() == 0 {
            return 0;
        }
        
        let mut hs: HashSet<i32> = nums.iter().cloned().collect();
        
        hs.iter().fold(0, |ans, n| {
            let mut curr = *n;
            if !hs.contains(&(curr-1)) {
                while hs.contains(&(curr+1)) {
                    curr += 1;
                }
            }
            
            ans.max(curr - n + 1)
        })
    }
}
```
