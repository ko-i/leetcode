char * addBinary(char * a, char * b){
    int a_len = strlen(a);
    int b_len = strlen(b);
    
    int result_len = (a_len > b_len ? a_len : b_len) + 2;
    char* result = (char*)malloc(result_len * sizeof(char));
    
    char* pr = &result[result_len - 1];
    *pr = '\0';
    pr--;
    
    int add = 0;
    int temp;
    while(a_len - 1 >= 0 && b_len - 1 >= 0)
    {
        temp = a[a_len - 1] - '0' + b[b_len - 1] - '0' + add;
        if(temp == 0)
        {
            *pr = '0';
            add = 0;
        }
        if(temp == 1)
        {
            *pr = '1';
            add = 0;
        }
        if(temp == 2)
        {
            *pr = '0';
            add = 1;
        }
        if(temp == 3)
        {
            *pr = '1';
            add = 1;
        }
        
        a_len--;
        b_len--;
        pr--;
    }
    
    while(a_len - 1 >= 0)
    {
        temp = a[a_len - 1] - '0' + add;
        if(temp == 0)
        {
            *pr = '0';
            add = 0;
        }
        if(temp == 1)
        {
            *pr = '1';
            add = 0;
        }
        if(temp == 2)
        {
            *pr = '0';
            add = 1;
        }
        
        a_len--;
        pr--;
    }
    
    while(b_len - 1 >= 0)
    {
        temp = b[b_len - 1] - '0' + add;
        if(temp == 0)
        {
            *pr = '0';
            add = 0;
        }
        if(temp == 1)
        {
            *pr = '1';
            add = 0;
        }
        if(temp == 2)
        {
            *pr = '0';
            add = 1;
        }
        
        b_len--;
        pr--;
    }
    
    if(add == 1)
    {
        *pr = '1';
        return pr;
    }
    
    return pr + 1;
}
