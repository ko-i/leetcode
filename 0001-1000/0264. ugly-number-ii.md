use std::cmp;

impl Solution {
    pub fn nth_ugly_number(n: i32) -> i32 {
        let n = n as usize;
        let mut dp = vec![0; n + 1];
        dp[1] = 1;
        
        let mut p2 = 1;
        let mut p3 = 1;
        let mut p5 = 1;
        
        for i in 2 ..= n {
            let num2 = dp[p2] * 2;
            let num3 = dp[p3] * 3;
            let num5 = dp[p5] * 5;
            
            dp[i] = cmp::min(cmp::min(num2, num3), num5);
            
            if dp[i] == num2 {
                p2 += 1;
            }
            
            if dp[i] == num3 {
                p3 += 1;
            }
            
            if dp[i] == num5 {
                p5 += 1;
            }
        }
        
        dp[n]
    }
}
