> https://leetcode.com/problems/counting-bits/

``` c
/**
 * Note: The returned array must be malloced, assume caller calls free().
 */
int* countBits(int num, int* returnSize){
    int* result = (int*)malloc((num + 1) * sizeof(int));
    *returnSize = (num + 1);
    
    for(int i = 0; i <= num; i++)
    {
        int temp = i;
        int one_count = 0;
        while(temp != 0)
        {
            if(temp & 1 == 1)
            {
                one_count++;
            }
            temp >>= 1;
        }
        
        result[i] = one_count;
    }
    
    return result;
}
```
