impl Solution {
    pub fn eval_rpn(tokens: Vec<String>) -> i32 {
        let mut nums_s = Vec::with_capacity(128);
        tokens.iter().for_each(|t| {
            match t.as_str() {
                "+" => {
                    let num2 = nums_s.pop().unwrap();
                    let num1 = nums_s.pop().unwrap();
                    nums_s.push(num1 + num2);
                },
                "-" => {
                    let num2 = nums_s.pop().unwrap();
                    let num1 = nums_s.pop().unwrap();
                    nums_s.push(num1 - num2);
                },
                "*" => {
                    let num2 = nums_s.pop().unwrap();
                    let num1 = nums_s.pop().unwrap();
                    nums_s.push(num1 * num2);
                },
                "/" => {
                    let num2 = nums_s.pop().unwrap();
                    let num1 = nums_s.pop().unwrap();
                    nums_s.push(num1 / num2);
                },
                _ => nums_s.push(t.parse().unwrap()),
            }
        });
        
        nums_s[0]
    }
}
