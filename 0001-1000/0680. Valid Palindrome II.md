bool validPalindrome(char * s){
    char* left = s;
    char* right = &s[strlen(s) - 1];
    
    while(left < right)
    {
        if(*left != *right)
        {
            break;
        }
        
        left++;
        right--;
    }
    
    char* second_left = left + 1;
    char* second_right = right;
    while(second_left < second_right)
    {
        if(*second_left != *second_right)
        {
            break;
        }
        
        second_left++;
        second_right--;
    }
    
    if(second_left >= second_right)
    {
        return true;
    }
    
    second_left = left;
    second_right = right - 1;
    while(second_left < second_right)
    {
        if(*second_left != *second_right)
        {
            return false;
        }
        
        second_left++;
        second_right--;
    }
    
    return true;
}
