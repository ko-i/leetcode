impl Solution {
    pub fn single_non_duplicate(nums: Vec<i32>) -> i32 {
        let (mut l, mut r) = (0, nums.len() - 1);
        
        while l != r {
            let mid = (l + r) >> 1;
            
            if nums[mid] == nums[mid+1] {
                if (mid - 1 - l + 1) & 1 == 0 {
                    // 左边数字个数是偶数，那么答案就在右边
                    l = mid + 2;
                } else {
                    // 右边数字个数是偶数，那么答案就在左边
                    r = mid - 1;
                }
            } else if nums[mid] == nums[mid-1] {
                if (r - mid + 1 - 1) & 1 == 0 {
                    // 右边数字是偶数，答案就在左边
                    r = mid - 2;
                } else {
                    // 左边数字是偶数，答案就在右边
                    l = mid + 1;
                }
            } else {
                return nums[mid];
            }
        }
        
        nums[l]
    }
}
