impl Solution {
    pub fn oranges_rotting(mut grid: Vec<Vec<i32>>) -> i32 {
        // é¦–å…ˆæˆ‘ä»¬ç»Ÿè®¡å‡ºæ–°é²œðŸŠçš„æ€»æ•°
        let mut remain_cnt = grid.iter().fold(0, |ans, grid_row| ans + grid_row.iter().filter(|&&n| n == 1).count());
        
        // ç„¶åŽæˆ‘ä»¬æŠŠæ‰€æœ‰è…çƒ‚çš„æ©˜å­åæ ‡ä½œä¸ºèµ·ç‚¹æ”¾å…¥å¹¿æœé˜Ÿåˆ—ä¸­(ç”¨ä¸¤ä¸ªæ•°ç»„æ¨¡æ‹Ÿ)
        let mut cur_set = Vec::with_capacity(grid.len() * 2);
        (0..grid.len()).for_each(|i| (0..grid[i].len()).for_each(|j| if grid[i][j] == 2 {cur_set.push((i, j))}));
        
        let mut ans = 0;
        while remain_cnt != 0 && !cur_set.is_empty() {
            let mut next_set = Vec::with_capacity(cur_set.len() * 2);
            cur_set.into_iter().for_each(|(r, c)| {
                if r > 0 && grid[r-1][c] == 1 {
                    grid[r-1][c] = 2;
                    remain_cnt -= 1;
                    next_set.push((r-1, c));
                }
            
                if r < grid.len() - 1 && grid[r+1][c] == 1 {
                    grid[r+1][c] = 2;
                    remain_cnt -= 1;
                    next_set.push((r+1, c));
                }
                
                if c > 0 && grid[r][c-1] == 1 {
                    grid[r][c-1] = 2;
                    remain_cnt -= 1;
                    next_set.push((r, c-1));
                }
            
                if c < grid[r].len() - 1 && grid[r][c+1] == 1 {
                    grid[r][c+1] = 2;
                    remain_cnt -= 1;
                    next_set.push((r, c+1));
                }
            });
            
            cur_set = next_set;
            ans += 1;
        }
        
        match remain_cnt == 0 {
            true => ans,
            _ => -1,
        }
    }
}
