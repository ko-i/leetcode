> https://leetcode.com/problems/relative-ranks/

impl Solution {
    pub fn find_relative_ranks(score: Vec<i32>) -> Vec<String> {
        let mut score_2_idx = (0..score.len()).map(|i| (score[i], i)).collect::<Vec<(i32, usize)>>();
        score_2_idx.sort_unstable_by(|a, b| b.cmp(&a));
        
        let mut res = vec![String::default(); score.len()];
        (0..score_2_idx.len()).for_each(|i| match i {
            0 => res[score_2_idx[0].1] = "Gold Medal".to_string(),
            1 => res[score_2_idx[1].1] = "Silver Medal".to_string(),
            2 => res[score_2_idx[2].1] = "Bronze Medal".to_string(),
            _ => res[score_2_idx[i].1] = (i+1).to_string(),
        });
        
        res
    }
}
