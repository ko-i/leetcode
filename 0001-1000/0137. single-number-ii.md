``` rust
impl Solution {
    pub fn single_number(nums: Vec<i32>) -> i32 {
        // 出现1次的位
        let mut once: i32 = 0;
        // 出现2次的位
        let mut twice: i32 = 0;
        
        for num in nums {
            // 第二次出现的位更新为：出现了一次的位与当前数字出现的位
            twice |= once & num;
            // 第一次出现的位更新为：0变1；而1变零表示当前位出现了两次，所以要清零
            once ^= num;
            
            // 计算当前出现三次的位
            let three = once & twice;
            
            // 将出现三次的位置零
            once &= !three;
            twice &= !three;
        }
        
        // 最后返回只出现了1次的位
        once
    }
}
```
