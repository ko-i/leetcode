``` c
#define QUEUE_SIZE 8192

struct Node* connect(struct Node* root) {
	if(!root) return root;
    
    struct Node** q = malloc(QUEUE_SIZE * sizeof(struct Node*));
    q[0] = root;
    int l = 0, r = 0, tail = 1;
    while(l <= r)
    {
        while(l <= r)
        {
            if(q[l] -> left) q[tail++] = q[l] -> left;
            if(q[l] -> right) q[tail++] = q[l] -> right;
            
            if(l < r) q[l] -> next = q[l + 1];
            else q[l] -> next = NULL;
            l++;
        }
        
        r = tail - 1;
    }
    
    free(q);
    
    return root;
}
```
