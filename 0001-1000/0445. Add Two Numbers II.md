``` c
#define MAX_SIZE 1024

struct ListNode* addTwoNumbers(struct ListNode* l1, struct ListNode* l2){
    int* s1 = malloc(MAX_SIZE * sizeof *s1);
    int* s2 = malloc(MAX_SIZE * sizeof *s2);
    int s1_top = 0, s2_top = 0;
    while(l1) {
        s1[s1_top++] = l1 -> val;
        l1 = l1 -> next;
    }
    
    while(l2) {
        s2[s2_top++] = l2 -> val;
        l2 = l2 -> next;
    }
    
    int carry = 0;
    struct ListNode* curr = NULL;
    while(s1_top && s2_top) {
        struct ListNode* nn = malloc(sizeof *nn);
        int sum = s1[s1_top-1] + s2[s2_top-1] + carry;
        nn -> val = sum % 10;
        nn -> next = curr;
        curr = nn;
        carry = sum / 10;
        s1_top--;
        s2_top--;
    }
    
    int* remain_s = NULL;
    int remain_top = 0;
    if(s1_top) {
        remain_s = s1;
        remain_top = s1_top;
    } else {
        remain_s = s2;
        remain_top = s2_top;
    }
    
    while(remain_top) {
        struct ListNode* nn = malloc(sizeof *nn);
        int sum = remain_s[remain_top-1] + carry;
        nn -> val = sum % 10;
        nn -> next = curr;
        curr = nn;
        carry = sum / 10;
        remain_top--;
    }
    
    if(carry) {
        struct ListNode* nn = malloc(sizeof *nn);
        nn -> val = carry;
        nn -> next = curr;
        curr = nn;
    }
    
    free(s1);
    free(s2);
    
    return curr;
}
```
