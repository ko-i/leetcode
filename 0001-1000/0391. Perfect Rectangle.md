use std::collections::HashMap;

impl Solution {
    pub fn is_rectangle_cover(rectangles: Vec<Vec<i32>>) -> bool {
        let mut point_2_cnt = HashMap::new();
        let mut edge = [i32::MAX, i32::MAX, i32::MIN, i32::MIN];
        
        if rectangles.iter().fold(0, |area, rect| {
            (0..2).for_each(|i| edge[i] = edge[i].min(rect[i]));
            (2..4).for_each(|i| edge[i] = edge[i].max(rect[i]));
            
            (0..3).step_by(2).for_each(|i| (1..4).step_by(2).for_each(|j| *point_2_cnt.entry((rect[i], rect[j])).or_insert(0) += 1));
            
            area + (rect[2] - rect[0]) * (rect[3] - rect[1])
        }) != (edge[2] - edge[0]) * (edge[3] - edge[1]) {
            return false;
        }
        
        match (point_2_cnt.remove(&(edge[0], edge[1])), point_2_cnt.remove(&(edge[0], edge[3])), point_2_cnt.remove(&(edge[2], edge[1])), point_2_cnt.remove(&(edge[2], edge[3]))) {
            (Some(1), Some(1), Some(1), Some(1)) => (),
            _ => return false,
        };
        
        point_2_cnt.iter().all(|(_, &v)| v == 2 || v == 4)
    }
}
