use std::collections::HashMap;

impl Solution {
    pub fn is_rectangle_cover(rectangles: Vec<Vec<i32>>) -> bool {
        let mut point_2_cnt = HashMap::new();
        let (mut left, mut right, mut bottom, mut top) = (i32::MAX, i32::MIN, i32::MAX, i32::MIN);
        let area = rectangles.iter().fold(0, |area, rect| {
            left = left.min(rect[0]);
            bottom = bottom.min(rect[1]);
            right = right.max(rect[2]);
            top = top.max(rect[3]);
            
            *point_2_cnt.entry((rect[0], rect[1])).or_insert(0) += 1;
            *point_2_cnt.entry((rect[0], rect[3])).or_insert(0) += 1;
            *point_2_cnt.entry((rect[2], rect[1])).or_insert(0) += 1;
            *point_2_cnt.entry((rect[2], rect[3])).or_insert(0) += 1;
            
            area + (rect[2] - rect[0]) * (rect[3] - rect[1])
        });
        
        if area != (right - left) * (top - bottom) {
            return false;
        }
        
        match (point_2_cnt.remove(&(left, bottom)), point_2_cnt.remove(&(left, top)), point_2_cnt.remove(&(right, bottom)), point_2_cnt.remove(&(right, top))) {
            (Some(1), Some(1), Some(1), Some(1)) => (),
            _ => return false,
        };
        
        point_2_cnt.iter().all(|(_, &v)| v == 2 || v == 4)
    }
}
