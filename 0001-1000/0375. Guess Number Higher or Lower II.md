impl Solution {
    pub fn get_money_amount(n: i32) -> i32 {
        let mut cache = vec![vec![0x3f3f3f3f; n as usize + 1]; n as usize + 1];
        Solution::dfs(1, n as usize, &mut cache)
    }
    
    fn dfs(l: usize, r: usize, cache: &mut Vec<Vec<i32>>) -> i32 {
        if l >= r {
            return 0;
        }
        
        if cache[l][r] != 0x3f3f3f3f {
            return cache[l][r];
        }
        
       (l..=r).for_each(|i| cache[l][r] = cache[l][r].min(i as i32 + Solution::dfs(l, i-1, cache).max(Solution::dfs(i+1, r, cache))));
       cache[l][r]
    } 
}

impl Solution {
    pub fn get_money_amount(n: i32) -> i32 {
        let mut dp = vec![vec![0x3f3f3f3f; n as usize + 1]; n as usize + 1];
        (1..dp.len()).for_each(|i| dp[i][i] = 0);
        (1..dp.len()-1).for_each(|i| dp[i][i+1] = i as i32);
        (1..dp.len()-2).rev().for_each(|i| 
            (i+2..dp[i].len()).for_each(|j| {
                dp[i][j] = dp[i][j].min(i as i32 + dp[i+1][j]);
                (i+1..j).for_each(|k| {
                    dp[i][j] = dp[i][j].min(dp[i][k-1].max(dp[k+1][j]) + k as i32);
                });
                dp[i][j] = dp[i][j].min(j as i32 + dp[i][j-1]);
            })
        );
        
        dp[1][n as usize]
    }
}
