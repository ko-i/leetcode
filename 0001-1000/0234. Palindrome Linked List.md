bool isPalindrome(struct ListNode* head){
    struct ListNode* p = head;
    int count = 0;
    while(p != NULL)
    {
        count++;
        p = p -> next;
    }
    
    if(count == 0 || count == 1)
    {
        return true;
    }
    
    int mid = count / 2 + 1;
    
    p = head;
    struct ListNode* p_front = NULL;
    struct ListNode* p_next;
    while(mid > 1)
    {
        p_next = p -> next;
        p -> next = p_front;
        p_front = p;
        p = p_next;
        
        mid--;
    }
    
    if(count % 2 == 1)
    {
        p_next = p_next -> next;
    }
    
    while(p_front != NULL && p_next != NULL)
    {
        if(p_front -> val != p_next -> val)
        {
            return false;
        }
        
        p_front = p_front -> next;
        p_next = p_next -> next;
    }
    
    return true;
}
