impl Solution {
    pub fn min_path_sum(grid: Vec<Vec<i32>>) -> i32 {
        let mut dp = grid;
        
        let row_size = dp.len();
        let col_size = dp[0].len();
        
        for j in 1..col_size {
            dp[0][j] += dp[0][j - 1];
        }
        
        for i in 1..row_size {
            dp[i][0] += dp[i - 1][0];
        }
        
        for i in 1..row_size {
            for j in 1..col_size {
                dp[i][j] += dp[i - 1][j].min(dp[i][j - 1]);
            }
        }
        
        dp[row_size - 1][col_size - 1]
    }
}
