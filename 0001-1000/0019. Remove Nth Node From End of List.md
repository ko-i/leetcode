struct ListNode* removeNthFromEnd(struct ListNode* head, int n){
    struct ListNode* p_front = head;
    struct ListNode* p = head;
    struct ListNode* p_next_n = head;
    for(int i = 0; i < n; i++)
    {
        p_next_n = p_next_n -> next;
    }
    
    if(p_next_n != NULL)
    {
        p = p -> next;
        p_next_n = p_next_n -> next;
    }
    
    while(p_next_n != NULL)
    {
        p_front = p;
        p = p -> next;
        p_next_n = p_next_n -> next;
    }
    
    if(p_front == p)
    {
        return head -> next;
    }
    
    p_front -> next = p -> next;
    
    return head;
}
