int pivotIndex(int* nums, int numsSize){
    int* right_to_left_sum = (int*)malloc(numsSize * sizeof(int));
    
    int sum = 0;
    for(int i = numsSize - 1; i >= 0; i--)
    {
        right_to_left_sum[i] = sum;
        sum += nums[i];
    }
    
    sum = 0;
    int temp;
    for(int i = 0; i < numsSize; i++)
    {
        temp = nums[i];
        nums[i] = sum;
        
        if(nums[i] == right_to_left_sum[i])
        {
            return i;
        }
        sum += temp; 
    }
    
    return -1;
}
