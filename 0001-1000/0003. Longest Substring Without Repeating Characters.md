int lengthOfLongestSubstring(char * s){
    if(strlen(s) == 0)
    {
        return 0;
    }
    
    int result = 1;
    char* left = s;
    char* right = left + 1;
    
    while(left != NULL && *left != '\0' && *right != '\0')
    {
        int not_same_count = 1;
        right = left + 1;
        while(*right != '\0')
        {
            char* temp_left = left;
            int same_flag = 0;
            
            while(temp_left < right)
            {
                if(*temp_left == *right)
                {
                    same_flag = 1;
                    left = temp_left + 1;
                    break;
                }
                
                temp_left++;
            }
            
            if(same_flag == 1)
            {
                break;
            }
            
            not_same_count++;
            
            if(not_same_count > result)
            {
                result = not_same_count;
            }
            
            right++;
        }
        
        if(*right == '\0')
        {
            break;
        }
    }
    
    return result;
}
