bool isPalindrome(char * s){
    char* left = s;
    char* right = &s[strlen(s) - 1];
    
    while(left < right)
    {
        while(left < right && !isalnum(*left))
        {
            left++;
        }
        
        if(left == right)
        {
            break;
        }
        
        while(right > left && !isalnum(*right))
        {
            right--;
        }
        
        if(right == left)
        {
            break;
        }
        
        if(tolower(*left) != tolower(*right))
        {
            return false;
        }
        
        left++;
        right--;
    }
    
    return true;
}
