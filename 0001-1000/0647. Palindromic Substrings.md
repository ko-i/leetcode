int paliCount(char* s, int s_len, int left, int right)
{
    int ans = 0;
    
    while(left >= 0 && right < s_len && s[left] == s[right])
    {
        ans++;
        
        left--;
        right++;
    }
    
    return ans;
}

int countSubstrings(char * s){
    int s_len = strlen(s);
    
    int ans = 0;
    
    for(int i = 0; i < s_len; i++)
    {
        ans += paliCount(s, s_len, i, i);
        
        if(s[i] == s[i + 1])
        {
            ans += paliCount(s, s_len, i, i + 1);
        }
    }
    
    return ans;
}
