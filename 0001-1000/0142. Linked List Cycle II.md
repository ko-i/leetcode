struct ListNode *detectCycle(struct ListNode *head) {
    if(!head || !head->next) return NULL;
    
    struct ListNode* s = head;
    struct ListNode* q = head;
    
    do {
        s = s->next;
        q = q->next->next;
    } while(s && q && q->next && s != q);
    
    if(!s || !q || !q->next) return NULL;
    
    q = head;
    while(s != q) {
        s = s->next;
        q = q->next;
    }
    
    return s;
}
