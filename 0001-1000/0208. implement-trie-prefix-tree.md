#define LETTER_SIZE 26

typedef struct trie{
    char letter;
    struct trie** links;
    bool isEnd;
} Trie;

/** Initialize your data structure here. */

Trie* newNode(char le)
{
    Trie* obj = calloc(1, sizeof(Trie));
    obj -> letter = le;
    obj -> links = calloc(LETTER_SIZE, sizeof(Trie*));
    obj -> isEnd = false;
    
    return obj;
}

Trie* trieCreate() {
    return newNode(0);
}

/** Inserts a word into the trie. */
void trieInsert(Trie* obj, char * word) {
    Trie* cur = obj;
    char* p_w = word;
    while(*p_w != '\0')
    {
        if(!cur -> links[*p_w - 'a']) cur -> links[*p_w - 'a'] = newNode(*p_w);
        
        cur = cur -> links[*p_w++ - 'a'];
    }
    
    cur -> isEnd = true;
}

/** Returns if the word is in the trie. */
bool trieSearch(Trie* obj, char * word) {
    Trie* cur = obj;
    char* p_w = word;
    while(*p_w != '\0')
    {
        if(!cur -> links[*p_w - 'a']) return false;
        
        cur = cur -> links[*p_w++ - 'a'];
    }
    
    return cur -> isEnd;
}

/** Returns if there is any word in the trie that starts with the given prefix. */
bool trieStartsWith(Trie* obj, char * prefix) {
    Trie* cur = obj;
    char* p_w = prefix;
    while(*p_w != '\0')
    {
        if(!cur -> links[*p_w - 'a']) return false;
        
        cur = cur -> links[*p_w++ - 'a'];
    }
    
    return true;
}

void trieFree(Trie* obj) {
    for(int i = 0; i < LETTER_SIZE; i++) if(obj -> links[i]) trieFree(obj -> links[i]);
    
    free(obj -> links);
    free(obj);
}
