/**
 * Note: The returned array must be malloced, assume caller calls free().
 */
int* getNoZeroIntegers(int n, int* returnSize){
    int* result = (int*)malloc(2 * sizeof(int));
    
    for(int i = 1; i <= n / 2; i++)
    {
        int temp_a = i;
        int temp_b = n - i;
        
        bool a_valid = true;
        bool b_valid = true;
        
        while(temp_a != 0)
        {
            if(temp_a % 10 == 0)
            {
                a_valid = false;
                break;
            }
            temp_a /= 10;
        }
        
        if(!a_valid)
        {
            continue;
        }
        
        while(temp_b != 0)
        {
            if(temp_b % 10 == 0)
            {
                b_valid = false;
                break;
            }
            temp_b /= 10;
        }
        
        if(!b_valid)
        {
            continue;
        }
        
        result[0] = i;
        result[1] = n - i;
        
        break; 
    }
    
    *returnSize = 2;
    return result;
}
